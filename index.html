Sample for an asus router:

#!/bin/sh
cd /tmp/mnt/USBDriveName/asusware
rm hosts.txt
wget http://mrdamien.github.io/dnsmasq-blocklist/hosts.txt
killall dnsmasq
echo "" > /tmp/etc/dnsmasq.conf
echo "pid-file=/var/run/dnsmasq.pid" >> /tmp/etc/dnsmasq.conf
echo "user=nobody" >> /tmp/etc/dnsmasq.conf
echo "bind-dynamic" >> /tmp/etc/dnsmasq.conf
echo "interface=br0" >> /tmp/etc/dnsmasq.conf
echo "interface=ppp1*" >> /tmp/etc/dnsmasq.conf
echo "no-dhcp-interface=ppp1*" >> /tmp/etc/dnsmasq.conf
echo "resolv-file=/tmp/resolv.conf" >> /tmp/etc/dnsmasq.conf
echo "servers-file=/tmp/resolv.dnsmasq" >> /tmp/etc/dnsmasq.conf
echo "no-poll" >> /tmp/etc/dnsmasq.conf
echo "no-negcache" >> /tmp/etc/dnsmasq.conf
echo "cache-size=2000" >> /tmp/etc/dnsmasq.conf
echo "min-port=4096" >> /tmp/etc/dnsmasq.conf
echo "dhcp-range=lan,192.168.1.2,192.168.1.254,255.255.255.0,86400s" >> /tmp/etc/dnsmasq.conf
echo "dhcp-option=lan,3,192.168.1.1" >> /tmp/etc/dnsmasq.conf
echo "dhcp-option=lan,6,192.168.1.1,0.0.0.0" >> /tmp/etc/dnsmasq.conf
echo "dhcp-option=lan,44,192.168.1.1" >> /tmp/etc/dnsmasq.conf
echo "dhcp-option=lan,252,\"\n\"" >> /tmp/etc/dnsmasq.conf
echo "dhcp-authoritative" >> /tmp/etc/dnsmasq.conf
echo "read-ethers" >> /tmp/etc/dnsmasq.conf
cat hosts.txt >> /tmp/etc/dnsmasq.conf
dnsmasq &

Asus routers will run ./asusware/.asusrouter scripts on startup
